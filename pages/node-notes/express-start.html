const app = express()

  response.write('hi')
  next()
})

  response.write('hi')
  next()
})

app.use((request,response,next)=>{ 
  response.end()

# 中间件

  • 被插入到启动和结束的物件, 就是上面代码的 app.use()
  • 一个 app.use() 就是一个中间件

优点:

  1. 每个功能都能通过一个函数实现
  2. 然后通过 app.use 将这个函数整合起来
  3. 如果把函数放到文件或 npm 里, 就实现了模块化

# 实现路由
