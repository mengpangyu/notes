# 处理只能选中一个标签的逻辑,

toggle(tag: string) {
  if(this.selectedTags.length>=1 && tag!==this.selectedTags[0]){
    return ;
  }
  if(this.selectedTags.indexOf(tag)>=0){
    this.selectedTags.splice(0, 1);
  }else{
    this.selectedTags.push(tag);
  }
}
